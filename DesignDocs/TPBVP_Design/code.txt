// =====================================================================
// ASTRODYNAMICS CONSTANTS & STATE
// =====================================================================
const DU = 1.495978707e11;  
const TU = 5.0226428e6;     
const MU_SUN = 1.0;

// N-BODY ADDITIONS: We inflate the target mass slightly so the SOI is large enough to clearly see the orbit on screen.
const MU_TARGET = 3e-5; 
const MU_EARTH = 3e-6; // Earth's gravitational parameter

let EARTH_PHASE_RAD = 0;
let TARGET_PHASE_RAD = (180 * Math.PI) / 180;
let TRANSFER_DAYS = 90.0;
let TF = (TRANSFER_DAYS * 86400) / TU;

let TARGET_TYPE = 'comet'; 
let INSERT_ORBIT = true;
const ORBIT_RADIUS = 0.01; // AU target parking orbit
const EARTH_ORBIT_RADIUS = 0.005; // AU starting parking orbit

// Freefall orbital velocity mathematically dictated by our MU
const ORBIT_OMEGA = Math.sqrt(MU_TARGET / Math.pow(ORBIT_RADIUS, 3)); 
const EARTH_ORBIT_OMEGA = Math.sqrt(MU_EARTH / Math.pow(EARTH_ORBIT_RADIUS, 3));

// Comet Parameters (a=2.5 AU, e=0.6)
const COMET_A = 2.5;
const COMET_E = 0.6;
const COMET_N = Math.sqrt(MU_SUN / Math.pow(COMET_A, 3));
const COMET_ARG_PERI = (30 * Math.PI) / 180; 

// =====================================================================
// KEPLER SOLVER & EPHEMERIS
// =====================================================================
function solveKepler(M, e) {
    let E = M;
    for (let i = 0; i < 10; i++) {
        E = E - (E - e * Math.sin(E) - M) / (1 - e * Math.cos(E));
    }
    return E;
}

function getEarthState(t) {
    const r = 1.0, omega = 1.0, theta = omega * t + EARTH_PHASE_RAD;
    return [r * Math.cos(theta), r * Math.sin(theta), -r * omega * Math.sin(theta), r * omega * Math.cos(theta)];
}

function getEarthOrbitState(t) {
    const e = getEarthState(t);
    const phase = EARTH_ORBIT_OMEGA * t;
    return [
        e[0] + EARTH_ORBIT_RADIUS * Math.cos(phase),
        e[1] + EARTH_ORBIT_RADIUS * Math.sin(phase),
        e[2] - EARTH_ORBIT_RADIUS * EARTH_ORBIT_OMEGA * Math.sin(phase),
        e[3] + EARTH_ORBIT_RADIUS * EARTH_ORBIT_OMEGA * Math.cos(phase)
    ];
}

function getMarsState(t) {
    const r = 1.524, omega = Math.pow(r, -1.5), theta = omega * t + TARGET_PHASE_RAD;
    return [r * Math.cos(theta), r * Math.sin(theta), -r * omega * Math.sin(theta), r * omega * Math.cos(theta)];
}

function getCometState(t) {
    const M = COMET_N * t + TARGET_PHASE_RAD;
    const E = solveKepler(M, COMET_E);
    
    const x_prime = COMET_A * (Math.cos(E) - COMET_E);
    const y_prime = COMET_A * Math.sqrt(1 - COMET_E * COMET_E) * Math.sin(E);
    
    const Edot = COMET_N / (1 - COMET_E * Math.cos(E));
    const vx_prime = -COMET_A * Math.sin(E) * Edot;
    const vy_prime = COMET_A * Math.sqrt(1 - COMET_E * COMET_E) * Math.cos(E) * Edot;
    
    const cosw = Math.cos(COMET_ARG_PERI), sinw = Math.sin(COMET_ARG_PERI);
    return [
        x_prime * cosw - y_prime * sinw,
        x_prime * sinw + y_prime * cosw,
        vx_prime * cosw - vy_prime * sinw,
        vx_prime * sinw + vy_prime * cosw
    ];
}

function getPrimaryTargetState(t) {
    return TARGET_TYPE === 'mars' ? getMarsState(t) : getCometState(t);
}

function getTargetState(t) {
    const body = getPrimaryTargetState(t);
    if (!INSERT_ORBIT) return body;

    const phase = ORBIT_OMEGA * t;
    return [
        body[0] + ORBIT_RADIUS * Math.cos(phase),
        body[1] + ORBIT_RADIUS * Math.sin(phase),
        body[2] - ORBIT_RADIUS * ORBIT_OMEGA * Math.sin(phase),
        body[3] + ORBIT_RADIUS * ORBIT_OMEGA * Math.cos(phase)
    ];
}

// =====================================================================
// N-BODY MATH & BVP SOLVER UTILS
// =====================================================================

// Now includes "t" because target planet moves over time
function computeDerivatives(t, state) {
    const [x, y, vx, vy, lrx, lry, lvx, lvy] = state;
    
    // Softening parameters to prevent BVP singularities if guess crosses planet
    const r2 = x*x + y*y + 1e-6; 
    const r3 = Math.pow(r2, 1.5);
    const r5 = r3 * r2;

    const targetBody = getPrimaryTargetState(t);
    const tx = targetBody[0], ty = targetBody[1];
    const dx = x - tx, dy = y - ty;
    
    const d2 = dx*dx + dy*dy + 1e-6; 
    const d3 = Math.pow(d2, 1.5);
    const d5 = d3 * d2;

    const earthBody = getEarthState(t);
    const ex = earthBody[0], ey = earthBody[1];
    const dex = x - ex, dey = y - ey;
    
    const de2 = dex*dex + dey*dey + 1e-6; 
    const de3 = Math.pow(de2, 1.5);
    const de5 = de3 * de2;

    const ax = -lvx, ay = -lvy;
    const dx_dt = vx, dy_dt = vy;

    // N-Body Gravity (Sun + Target + Earth)
    const dvx = -MU_SUN * x / r3 - MU_TARGET * dx / d3 - MU_EARTH * dex / de3 + ax;
    const dvy = -MU_SUN * y / r3 - MU_TARGET * dy / d3 - MU_EARTH * dey / de3 + ay;

    // Costate Derivatives (N-Body Gravity Gradient Tensor)
    const dot_r_lv = x * lvx + y * lvy;
    const j_sun_x = MU_SUN * lvx / r3 - 3 * MU_SUN * x * dot_r_lv / r5;
    const j_sun_y = MU_SUN * lvy / r3 - 3 * MU_SUN * y * dot_r_lv / r5;

    const dot_d_lv = dx * lvx + dy * lvy;
    const j_tar_x = MU_TARGET * lvx / d3 - 3 * MU_TARGET * dx * dot_d_lv / d5;
    const j_tar_y = MU_TARGET * lvy / d3 - 3 * MU_TARGET * dy * dot_d_lv / d5;

    const dot_de_lv = dex * lvx + dey * lvy;
    const j_ear_x = MU_EARTH * lvx / de3 - 3 * MU_EARTH * dex * dot_de_lv / de5;
    const j_ear_y = MU_EARTH * lvy / de3 - 3 * MU_EARTH * dey * dot_de_lv / de5;

    const dlrx = j_sun_x + j_tar_x + j_ear_x;
    const dlry = j_sun_y + j_tar_y + j_ear_y;
    const dlvx = -lrx, dlvy = -lry;

    return [dx_dt, dy_dt, dvx, dvy, dlrx, dlry, dlvx, dlvy];
}

function rk4Step(t, state, dt) {
    const k1 = computeDerivatives(t, state);
    const s2 = state.map((v, i) => v + 0.5 * dt * k1[i]);
    const k2 = computeDerivatives(t + 0.5 * dt, s2);
    const s3 = state.map((v, i) => v + 0.5 * dt * k2[i]);
    const k3 = computeDerivatives(t + 0.5 * dt, s3);
    const s4 = state.map((v, i) => v + dt * k3[i]);
    const k4 = computeDerivatives(t + dt, s4);
    return state.map((v, i) => v + (dt / 6.0) * (k1[i] + 2*k2[i] + 2*k3[i] + k4[i]));
}

function integrateBVP(currentState, costates, t_now, t_final, steps=50, returnHistory=false) {
    let state = [...currentState, ...costates];
    let t = t_now;
    const dt = (t_final - t_now) / steps;
    const history = returnHistory ? [state.slice(0,4)] : null;
    for (let i = 0; i < steps; i++) {
        state = rk4Step(t, state, dt);
        t += dt;
        if (returnHistory) history.push(state.slice(0,4));
    }
    if (returnHistory) return history;
    const target = getTargetState(t_final); 
    return [state[0] - target[0], state[1] - target[1], state[2] - target[2], state[3] - target[3]];
}

function solveLinearSystem(A, b) {
    const n = b.length;
    let M = A.map((row, i) => [...row, b[i]]); 
    for (let i = 0; i < n; i++) {
        let maxRow = i;
        for (let k = i + 1; k < n; k++) if (Math.abs(M[k][i]) > Math.abs(M[maxRow][i])) maxRow = k;
        [M[i], M[maxRow]] = [M[maxRow], M[i]];
        for (let k = i + 1; k < n; k++) {
            const c = -M[k][i] / M[i][i];
            for (let j = i; j <= n; j++) M[k][j] = (i === j) ? 0 : M[k][j] + c * M[i][j];
        }
    }
    const x = new Array(n).fill(0);
    for (let i = n - 1; i >= 0; i--) {
        x[i] = M[i][n] / M[i][i];
        for (let k = i - 1; k >= 0; k--) M[k][n] -= M[k][i] * x[i];
    }
    return x;
}

function getKinematicGuess(r0, v0, t_now, t_final) {
    const rf = getTargetState(t_final), vf = [rf[2], rf[3]];
    const T = t_final - t_now;
    const dr = [rf[0] - r0[0], rf[1] - r0[1]];
    const dv = [vf[0] - v0[0], vf[1] - v0[1]];
    const c1x = (12 / Math.pow(T, 3)) * (dv[0]*T/2 - dr[0] + v0[0]*T);
    const c1y = (12 / Math.pow(T, 3)) * (dv[1]*T/2 - dr[1] + v0[1]*T);
    const c0x = dv[0]/T - c1x*T/2;
    const c0y = dv[1]/T - c1y*T/2;
    return [-c1x, -c1y, -c0x, -c0y]; 
}

function updateOrbitalElements() {
    if (!isRunning && simTime === 0) return;

    // Calculate relative Cartesian state against the center of the target body
    const target = getPrimaryTargetState(simTime);
    const rx = shipState[0] - target[0];
    const ry = shipState[1] - target[1];
    const vx = shipState[2] - target[2];
    const vy = shipState[3] - target[3];

    const r = Math.sqrt(rx*rx + ry*ry);
    const h = rx*vy - ry*vx; // Specific angular momentum (Z-component)
    
    const p = (h*h) / MU_TARGET; // Semi-latus rectum
    
    // Eccentricity vector components
    const ex = (vy*h)/MU_TARGET - rx/r;
    const ey = (-vx*h)/MU_TARGET - ry/r;
    const e = Math.sqrt(ex*ex + ey*ey);
    
    // Keplerian Elements
    const rp = p / (1 + e);
    const ra = (e < 1.0) ? (p / (1 - e)) : Infinity;
    const a = (e !== 1.0) ? (p / (1 - e*e)) : Infinity;
    
    let aop = Math.atan2(ey, ex) * 180 / Math.PI;
    if (aop < 0) aop += 360;
    
    let ta = 0;
    if (e > 1e-6) {
        let cos_nu = (ex*rx + ey*ry) / (e*r);
        if (cos_nu > 1) cos_nu = 1; // Clamp for floating point errors
        if (cos_nu < -1) cos_nu = -1;
        ta = Math.acos(cos_nu) * 180 / Math.PI;
        // Check if moving towards or away from periapsis
        if ((rx*vx + ry*vy) < 0) {
            ta = 360 - ta;
        }
    }

    const inc = h < 0 ? 180.0 : 0.0; // Retrograde vs Prograde in 2D plane

    document.getElementById('oe_sma').innerText = (a === Infinity || a < 0) ? (a < 0 ? a.toFixed(3) : "∞") + " AU" : a.toFixed(4) + " AU";
    document.getElementById('oe_ecc').innerText = e.toFixed(4);
    document.getElementById('oe_inc').innerText = inc.toFixed(1) + "°";
    document.getElementById('oe_aop').innerText = aop.toFixed(1) + "°";
    document.getElementById('oe_ta').innerText = ta.toFixed(1) + "°";
    document.getElementById('oe_rp').innerText = rp.toFixed(4) + " AU";
    document.getElementById('oe_ra').innerText = ra === Infinity ? "∞ AU" : ra.toFixed(4) + " AU";
}

function runShootingMethod(maxIters=3) {
    const eps = 1e-5, tolerance = 1e-5;
    let errMag = 999, iters = 0;
    
    for (let i = 0; i < maxIters; i++) {
        iters++;
        const errVec = integrateBVP(shipState, currentCostates, simTime, TF);
        errMag = Math.sqrt(errVec.reduce((s, e) => s + e*e, 0));
        if (errMag < tolerance) break;
        
        const J = [];
        for (let j = 0; j < 4; j++) {
            const pertC = [...currentCostates];
            pertC[j] += eps;
            const pErr = integrateBVP(shipState, pertC, simTime, TF);
            J.push(pErr.map((pe, k) => (pe - errVec[k]) / eps));
        }
        const J_T = [[],[],[],[]];
        for(let r=0; r<4; r++) for(let c=0; c<4; c++) J_T[r][c] = J[c][r];
        const delta = solveLinearSystem(J_T, errVec);
        for (let j = 0; j < 4; j++) currentCostates[j] -= 0.8 * delta[j]; 
    }
    document.getElementById('iterText').innerText = iters;
    document.getElementById('errorText').innerText = errMag.toFixed(4) + " AU";
    return errMag;
}

function guidanceTick() {
    if (simTime >= TF) return;

    const tgo_days = (TF - simTime) * TU / 86400;
    if (tgo_days <= TERMINAL_HANDOFF_DAYS && mode !== "TERMINAL") {
        mode = "TERMINAL";
        document.getElementById('modeText').innerText = "TERMINAL";
        document.getElementById('modeText').className = "text-yellow-400 font-bold terminal-glow";
        log("BVP Singularity imminent. Switching to Terminal Guidance.", "warn");
        projectedPath = []; 
    }

    if (mode === "OPTIMAL") {
        runShootingMethod(3); 
        projectedPath = integrateBVP(shipState, currentCostates, simTime, TF, 40, true);
    } 
}

function physicsStep(dt) {
    if (mode === "OPTIMAL") {
        // High fidelity N-Body RK4 Step including Costate propagation
        const fullState = [...shipState, ...currentCostates];
        const nextState = rk4Step(simTime, fullState, dt);
        for(let i=0; i<4; i++) shipState[i] = nextState[i];
        for(let i=0; i<4; i++) currentCostates[i] = nextState[i+4];
        
        const ax = -currentCostates[2];
        const ay = -currentCostates[3];
        const accel_mag = Math.sqrt(ax*ax + ay*ay) * (DU / (TU*TU));
        document.getElementById('thrustText').innerText = accel_mag.toFixed(3) + " m/s²";
    } else {
        // TERMINAL OR ORBITING
        // RK4 specifically for the ship state under PD Control + N-Body Gravity
        const rk4Ship = (t_eval, s, stepSize) => {
            const shipDerivs = (t, st) => {
                const target = getTargetState(t);
                const r_err = [target[0] - st[0], target[1] - st[1]];
                const v_err = [target[2] - st[2], target[3] - st[3]];
                
                // Terminal & Station Keeping Thrusters
                // Using critically damped high gains to tightly track the accelerating orbital setpoint
                const Kp = 10000.0, Kd = 200.0;
                let ax = Kp * r_err[0] + Kd * v_err[0];
                let ay = Kp * r_err[1] + Kd * v_err[1];
                
                // Cap terminal thrust to ~3.0 m/s^2 to prevent physics instability on large errors
                const max_a = 500.0;
                const a_mag = Math.sqrt(ax*ax + ay*ay);
                if (a_mag > max_a) {
                    ax = (ax / a_mag) * max_a;
                    ay = (ay / a_mag) * max_a;
                }
                
                // N-Body Gravity
                const r3 = Math.pow(st[0]*st[0] + st[1]*st[1] + 1e-6, 1.5);
                const tBody = getPrimaryTargetState(t);
                const dx = st[0] - tBody[0], dy = st[1] - tBody[1];
                const d3 = Math.pow(dx*dx + dy*dy + 1e-6, 1.5);
                
                const eBody = getEarthState(t);
                const dex = st[0] - eBody[0], dey = st[1] - eBody[1];
                const de3 = Math.pow(dex*dex + dey*dey + 1e-6, 1.5);
                
                const dvx = -MU_SUN * st[0] / r3 - MU_TARGET * dx / d3 - MU_EARTH * dex / de3 + ax;
                const dvy = -MU_SUN * st[1] / r3 - MU_TARGET * dy / d3 - MU_EARTH * dey / de3 + ay;
                
                return [st[2], st[3], dvx, dvy];
            };

            const k1 = shipDerivs(t_eval, s);
            const s2 = s.map((v, i) => v + 0.5 * stepSize * k1[i]);
            const k2 = shipDerivs(t_eval + 0.5 * stepSize, s2);
            const s3 = s.map((v, i) => v + 0.5 * stepSize * k2[i]);
            const k3 = shipDerivs(t_eval + 0.5 * stepSize, s3);
            const s4 = s.map((v, i) => v + stepSize * k3[i]);
            const k4 = shipDerivs(t_eval + stepSize, s4);
            return s.map((v, i) => v + (stepSize / 6.0) * (k1[i] + 2*k2[i] + 2*k3[i] + k4[i]));
        };
        
        shipState = rk4Ship(simTime, shipState, dt);
        
        // Calculate thrust for UI
        const target = getTargetState(simTime + dt);
        const r_err = [target[0] - shipState[0], target[1] - shipState[1]];
        const v_err = [target[2] - shipState[2], target[3] - shipState[3]];
        const Kp = 10000.0, Kd = 200.0;
        let ax = Kp * r_err[0] + Kd * v_err[0];
        let ay = Kp * r_err[1] + Kd * v_err[1];
        
        const max_a = 500.0;
        const a_mag = Math.sqrt(ax*ax + ay*ay);
        if (a_mag > max_a) {
            ax = (ax / a_mag) * max_a;
            ay = (ay / a_mag) * max_a;
        }
        
        const accel_mag = Math.sqrt(ax*ax + ay*ay) * (DU / (TU*TU));
        document.getElementById('thrustText').innerText = accel_mag.toFixed(3) + " m/s²";
    }
}

function simLoop(timestamp) {
    if (!isRunning) return;
    const realDt = (timestamp - realTimeLast) / 1000.0; 
    realTimeLast = timestamp;
    const simDaysToAdvance = realDt;
    const simTimeDelta = simDaysToAdvance * 86400 / TU;
    
    const physicsSteps = 10; 
    const physDt = simTimeDelta / physicsSteps;

    if (simTime < TF) {
        for (let i=0; i<physicsSteps; i++) {
            physicsStep(physDt);
            simTime += physDt;
            guidanceTimer += physDt;
            const guidanceInterval = (1.0 / GUIDANCE_HZ) * 86400 / TU;
            if (guidanceTimer >= guidanceInterval) {
                guidanceTick();
                guidanceTimer = 0;
                shipFlownPath.push([shipState[0], shipState[1]]); 
            }
        }
        const tgo = (TF - simTime) * TU / 86400;
        document.getElementById('tgoText').innerText = Math.max(0, tgo).toFixed(2) + " d";
    } else {
        if (mode !== "ORBITING") {
            mode = "ORBITING";
            document.getElementById('modeText').innerText = "ORBITING";
            document.getElementById('modeText').className = "text-purple-400 font-bold orbit-glow";
            document.getElementById('tgoText').innerText = "ARRIVED";
            document.getElementById('errorText').innerText = "0.000 AU";
            document.getElementById('iterText').innerText = "--";
            log(INSERT_ORBIT ? "Orbit insertion complete. N-Body gravity locked." : "Rendezvous complete.", "success");
        }

        for (let i=0; i<physicsSteps; i++) {
            physicsStep(physDt);
            simTime += physDt;
            shipFlownPath.push([shipState[0], shipState[1]]); 
            if (shipFlownPath.length > 300) shipFlownPath.shift(); 
        }
    }
    
    updateOrbitalElements();
    drawScene();
    if (isRunning) requestAnimationFrame(simLoop);
}

function drawOrbit(r, color) {
    ctx.beginPath(); ctx.arc(width/2, height/2, r * scale, 0, 2*Math.PI);
    ctx.strokeStyle = color; ctx.lineWidth = 1; ctx.setLineDash([4, 4]);
    ctx.stroke(); ctx.setLineDash([]);
}

function drawCometOrbit() {
    ctx.beginPath();
    const steps = 100;
    for(let i=0; i<=steps; i++) {
        const E = (i / steps) * 2 * Math.PI;
        const x_prime = COMET_A * (Math.cos(E) - COMET_E);
        const y_prime = COMET_A * Math.sqrt(1 - COMET_E * COMET_E) * Math.sin(E);
        
        const cosw = Math.cos(COMET_ARG_PERI), sinw = Math.sin(COMET_ARG_PERI);
        const x = x_prime * cosw - y_prime * sinw;
        const y = x_prime * sinw + y_prime * cosw;
        
        const sx = width/2 + x * scale;
        const sy = height/2 - y * scale;
        if(i===0) ctx.moveTo(sx, sy);
        else ctx.lineTo(sx, sy);
    }
    ctx.strokeStyle = 'rgba(236, 72, 153, 0.3)';
    ctx.lineWidth = 1; ctx.setLineDash([4, 4]);
    ctx.stroke(); ctx.setLineDash([]);
}

function drawScene() {
    ctx.clearRect(0, 0, width, height);
    
    // Sun
    ctx.beginPath(); ctx.arc(width/2, height/2, 8, 0, 2*Math.PI);
    ctx.fillStyle = '#fde047'; ctx.shadowColor = '#facc15'; ctx.shadowBlur = 10;
    ctx.fill(); ctx.shadowBlur = 0;

    drawOrbit(1.0, 'rgba(56, 189, 248, 0.2)');
    if (TARGET_TYPE === 'mars') {
        drawOrbit(1.524, 'rgba(251, 146, 60, 0.2)');
    } else {
        drawCometOrbit();
    }
    
    const e = getEarthState(simTime);
    const body = getPrimaryTargetState(simTime);
    
    // Draw Earth Starting Orbit Ring
    ctx.beginPath();
    ctx.arc(width/2 + e[0]*scale, height/2 - e[1]*scale, EARTH_ORBIT_RADIUS * scale, 0, 2*Math.PI);
    ctx.strokeStyle = 'rgba(56, 189, 248, 0.4)'; ctx.lineWidth = 1; ctx.stroke();

    // Draw Target Orbit Ring
    if (INSERT_ORBIT) {
        ctx.beginPath();
        ctx.arc(width/2 + body[0]*scale, height/2 - body[1]*scale, ORBIT_RADIUS * scale, 0, 2*Math.PI);
        ctx.strokeStyle = 'rgba(168, 85, 247, 0.3)'; ctx.lineWidth = 1; ctx.stroke();
    }

    if (!isRunning && simTime === 0) {
        const mTarget = getTargetState(TF);
        ctx.beginPath(); ctx.arc(width/2 + mTarget[0]*scale, height/2 - mTarget[1]*scale, 4, 0, 2*Math.PI); 
        ctx.strokeStyle = INSERT_ORBIT ? 'rgba(168, 85, 247, 0.8)' : (TARGET_TYPE==='mars' ? 'rgba(251, 146, 60, 0.5)' : 'rgba(236, 72, 153, 0.5)'); 
        ctx.lineWidth = 2; ctx.stroke();
        ctx.fillStyle = INSERT_ORBIT ? 'rgba(168, 85, 247, 0.8)' : (TARGET_TYPE==='mars' ? 'rgba(251, 146, 60, 0.8)' : 'rgba(236, 72, 153, 0.8)');
        ctx.font = '10px monospace';
        ctx.fillText(INSERT_ORBIT ? "Insertion Pt" : "Target", width/2 + mTarget[0]*scale + 8, height/2 - mTarget[1]*scale + 3);
    }

    ctx.beginPath(); ctx.arc(width/2 + e[0]*scale, height/2 - e[1]*scale, 3, 0, 2*Math.PI); ctx.fillStyle = '#38bdf8'; ctx.fill();
    ctx.beginPath(); ctx.arc(width/2 + body[0]*scale, height/2 - body[1]*scale, 4, 0, 2*Math.PI); 
    ctx.fillStyle = TARGET_TYPE === 'mars' ? '#fb923c' : '#ec4899'; 
    ctx.fill();

    if (shipFlownPath.length > 1) {
        ctx.beginPath();
        ctx.moveTo(width/2 + shipFlownPath[0][0]*scale, height/2 - shipFlownPath[0][1]*scale);
        for(let i=1; i<shipFlownPath.length; i++) {
            ctx.lineTo(width/2 + shipFlownPath[i][0]*scale, height/2 - shipFlownPath[i][1]*scale);
        }
        ctx.strokeStyle = mode === "ORBITING" ? 'rgba(168, 85, 247, 0.6)' : '#94a3b8'; 
        ctx.lineWidth = 2; ctx.stroke();
    }

    if (mode === "OPTIMAL" && projectedPath.length > 0) {
        ctx.beginPath();
        ctx.moveTo(width/2 + projectedPath[0][0]*scale, height/2 - projectedPath[0][1]*scale);
        for(let i=1; i<projectedPath.length; i++) {
            ctx.lineTo(width/2 + projectedPath[i][0]*scale, height/2 - projectedPath[i][1]*scale);
        }
        ctx.strokeStyle = 'rgba(34, 197, 94, 0.8)'; ctx.lineWidth = 2; ctx.setLineDash([5, 5]); ctx.stroke(); ctx.setLineDash([]);
    }

    const sx = width/2 + shipState[0]*scale, sy = height/2 - shipState[1]*scale;
    const ax = mode==="OPTIMAL" ? -currentCostates[2] : 0;
    const ay = mode==="OPTIMAL" ? -currentCostates[3] : 0;
    
    if (isRunning && mode !== "ORBITING") {
        ctx.beginPath(); ctx.moveTo(sx, sy); ctx.lineTo(sx - ax*2000, sy + ay*2000); 
        ctx.strokeStyle = mode==="OPTIMAL" ? '#ef4444' : '#facc15'; ctx.lineWidth = 3; ctx.stroke();
    }
    ctx.beginPath(); ctx.arc(sx, sy, 3, 0, 2*Math.PI); ctx.fillStyle = '#ffffff'; ctx.fill();
}

function resize() {
    width = canvas.parentElement.clientWidth;
    height = canvas.parentElement.clientHeight;

    scale = Math.min(width, height) / zoomFactor;
    drawScene();
}
